### Types

#### ButtonVariant

```ts
type ButtonVariant = "solid" | "bordered" | "flat" | "text";
```

#### Color

```ts
type Color =
  | "basic"
  | "primary"
  | "secondary"
  | "success"
  | "warning"
  | "error"
  | "gradient";
```

#### Size

```ts
type Size = "auto" | "xs" | "sm" | "md" | "lg" | "xl";
```

#### ButtonWidth

```ts
type ButtonWidth = "auto" | "full";
```

#### ButtonType

```ts
type ButtonType = "button" | "submit";
```
